```typescript
import { webFetchTool } from "../tools/web-fetch.js";
import { validateFetchedContent } from "../tools/validators/input-validator.js";
import { researchAgent } from "../agents/research-agent.js";
import { evaluateQualityAsync } from "../agents/quality-agent.js";
import { withRetry } from "../../utils/retry.js";

export interface ResearchPipelineInput {
  title: string;
  links: Array<{ url: string; key: string }>;
  document?: string;
}

export interface ResearchPipelineOutput {
  title: string;
  researchMarkdown: string;
  validationFailures?: Array<{ name: string; reason: string }>;
}

/**
 * Complete research pipeline: fetch -> validate -> research -> evaluate quality
 * Per D-09: 3 retries on failure, then fail with specific error
 * Per D-10: Input validation before passing to agent
 * Per D-11: Quality evaluation runs async, does not block
 */
export async function runResearchPipeline(
  input: ResearchPipelineInput,
  options: { maxRetries?: number } = {}
): Promise<ResearchPipelineOutput> {
  const maxRetries = options.maxRetries ?? 3;
  const retryOptions = { maxRetries, initialDelayMs: 1000, maxDelayMs: 5000, factor: 2 };

  // Step 1: Fetch all links
  const fetchedContent: Array<{ url: string; content: string; title: string }> = [];

  for (const link of input.links) {
    const fetchResult = await withRetry(
      async () => {
        const result = await webFetchTool.execute({ url: link.url });
        return result;
      },
      retryOptions
    );

    // Step 2: Validate fetched content (D-10)
    const validation = validateFetchedContent(fetchResult.content, fetchResult.url);

    if (!validation.valid) {
      if (maxRetries > 0) {
        // Retry with different approach or fail
        throw new Error(`Validation failed for ${link.url}: ${JSON.stringify(validation.failures)}`);
      } else {
        // Last attempt - record failures but continue
        fetchedContent.push({
          url: fetchResult.url,
          content: fetchResult.content,
          title: fetchResult.title,
        });
        continue;
      }
    }

    fetchedContent.push({
      url: fetchResult.url,
      content: fetchResult.content,
      title: fetchResult.title,
    });
  }

  // Step 3: Run research agent
  const researchResult = await researchAgent.run(
    `Title: ${input.title}\n\nLinks:\n${input.links.map((l, i) => `[${i + 1}] ${l.key}: ${l.url}`).join("\n")}\n\nContent:\n${fetchedContent.map((c, i) => `[${i + 1}] ${c.title}\n${c.content}`).join("\n\n")}`
  );

  const researchMarkdown = researchResult.text();

  // Step 4: Quality evaluation (non-blocking per D-11)
  evaluateQualityAsync(
    researchMarkdown,
    (score) => {
      console.log(`[Research Pipeline] Quality score: ${score.qualityScore}`);
    },
    (error) => {
      console.warn(`[Research Pipeline] Quality evaluation failed: ${error.message}`);
    }
  );

  // Pipeline continues immediately without waiting for quality evaluation

  return {
    title: input.title,
    researchMarkdown,
  };
}
```

**Key orchestration points:**
- D-09: Uses withRetry for fetch failures (3 retries)
- D-10: Validates content before research agent
- D-11: Quality evaluation runs async, does not block
- Sequential fetch -> validate -> research -> async quality

**Acceptance criteria:**
- Pipeline orchestrates fetch -> validate -> research -> quality
- withRetry handles fetch failures (3 retries per D-09)
- validateFetchedContent called before research agent
- Quality evaluation is fire-and-forget (does not block)
