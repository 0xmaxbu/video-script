```typescript
/**
 * Dark Mode Theme — Phase 16 Visual Polish
 *
 * Centralized palette for AI Jason aesthetic.
 * All layouts MUST import from here — no hardcoded colors allowed.
 *
 * NOTE: Remotion uses inline styles. Do NOT use Tailwind color utilities
 * for background/text colors (Tailwind utility classes work for layout/spacing only).
 * Animations always use useCurrentFrame() — never CSS transition-*/animate-* classes.
 */
export const THEME = {
  bg: {
    primary: "#0a0a0a",
    secondary: "#111111",
    card: "#1a1a1a",
  },
  text: {
    primary: "#ffffff",
    secondary: "rgba(255,255,255,0.7)",
    muted: "rgba(255,255,255,0.4)",
  },
  accent: {
    yellow: "#FFD700",
    yellowMuted: "rgba(255,215,0,0.3)",
    blue: "#3b82f6",
  },
  glass: {
    bg: "rgba(255,255,255,0.05)",
    border: "rgba(255,255,255,0.1)",
  },
} as const;

export type Theme = typeof THEME;
```
