```typescript
export const AnnotationSchema = z.object({
  type: AnnotationTypeEnum,  // "circle" | "underline" | "arrow" | "highlight" | "box" | "number"
  target: AnnotationTargetSchema,
  style: z.object({
    color: AnnotationColorEnum,
    size: z.enum(["small", "medium", "large"]),
  }),
  narrationBinding: z.object({
    triggerText: z.string(),
    segmentIndex: z.number().int().nonnegative(),
    appearAt: z.number().nonnegative(),
  }),
});
export type Annotation = z.infer<typeof AnnotationSchema>;
```

Note: Per D-07/D-08, region/coordinates positioning is used. For Phase 1, annotations use explicit x, y, radius/width/height props passed directly - the target.region field is ignored for now.
